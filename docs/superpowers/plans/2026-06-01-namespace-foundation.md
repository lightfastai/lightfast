# Namespace Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the durable Lightfast namespace foundation: shared handle validation, namespace tables, XState transition rules, and DB services for starting, reserving, finalizing, failing started operations, and deleting safe pre-Clerk reservations.

**Architecture:** Clerk remains outside this plan. This plan creates the Lightfast-owned namespace registry and operation state machine that later username/org flows will drive. XState defines legal transitions, but MySQL stores durable state and repository helpers perform all side effects explicitly.

**Tech Stack:** pnpm workspace catalogs, TypeScript, Zod, Drizzle ORM on PlanetScale/Vitess MySQL, Vitest, XState v5 pure transition API.

---

## Scope

This is Plan 1 only. Do not add username UI, account setup pages, Clerk user mutations, proxy routing changes, or backfill jobs here. Those come after this foundation lands.

This plan intentionally includes shared handle validation because the namespace DB unique key must receive normalized, route-safe handles from the start.

Reserved handle validation must use the existing `@repo/app-reserved-names` package as the route-reservation source of truth. Do not create a second reserved-name list in this plan.

```ts
import { organization } from "@repo/app-reserved-names";
import { z } from "zod";

const LIGHTFAST_HANDLE = {
  MIN_LENGTH: 4,
  MAX_LENGTH: 64,
  PATTERN: /^[a-z0-9-]+$/,
  START_PATTERN: /^[a-z0-9]/,
  END_PATTERN: /[a-z0-9]$/,
  NO_CONSECUTIVE_HYPHENS: /--/,
} as const;

const HANDLE_ERRORS = {
  MIN_LENGTH: `Handle must be at least ${LIGHTFAST_HANDLE.MIN_LENGTH} characters`,
  MAX_LENGTH: `Handle must be less than ${LIGHTFAST_HANDLE.MAX_LENGTH + 1} characters`,
  PATTERN: "Only lowercase letters, numbers, and hyphens are allowed",
  START: "Handle must start with a letter or number",
  END: "Handle must end with a letter or number",
  CONSECUTIVE: "Handle cannot contain consecutive hyphens",
  RESERVED: "This handle is reserved for system use. Please choose a different handle.",
} as const;

export function normalizeLightfastHandle(value: string) {
  return value.trim().toLowerCase();
}

export const lightfastHandleSchema = z.preprocess(
  (value) =>
    typeof value === "string" ? normalizeLightfastHandle(value) : value,
  z
    .string()
    .min(LIGHTFAST_HANDLE.MIN_LENGTH, HANDLE_ERRORS.MIN_LENGTH)
    .max(LIGHTFAST_HANDLE.MAX_LENGTH, HANDLE_ERRORS.MAX_LENGTH)
    .regex(LIGHTFAST_HANDLE.PATTERN, HANDLE_ERRORS.PATTERN)
    .regex(LIGHTFAST_HANDLE.START_PATTERN, HANDLE_ERRORS.START)
    .regex(LIGHTFAST_HANDLE.END_PATTERN, HANDLE_ERRORS.END)
    .refine((value) => !LIGHTFAST_HANDLE.NO_CONSECUTIVE_HYPHENS.test(value), {
      message: HANDLE_ERRORS.CONSECUTIVE,
    })
    .refine((handle) => !organization.check(handle), {
      message: HANDLE_ERRORS.RESERVED,
    })
);

export type LightfastHandle = z.infer<typeof lightfastHandleSchema>;
```

```ts
import { sql } from "drizzle-orm";
import {
  bigint,
  index,
  mysqlTable,
  timestamp,
  uniqueIndex,
  varchar,
} from "drizzle-orm/mysql-core";

export type NamespaceKind = "user" | "org";
export type NamespaceStatus = "reserved" | "active";
export type NamespaceOperationType =
  | "create_user_username"
  | "create_org_slug"
  | "backfill_existing_handle";
export type NamespaceOperationStatus =
  | "started"
  | "namespace_reserved"
  | "clerk_applied"
  | "finalized"
  | "failed"
  | "compensating";

const HANDLE_LENGTH = 64;
const CLERK_ID_LENGTH = 64;
const CODE_LENGTH = 48;
const IDEMPOTENCY_KEY_LENGTH = 128;
const ERROR_CODE_LENGTH = 64;
const ERROR_MESSAGE_LENGTH = 512;

export const namespaces = mysqlTable(
  "lightfast_namespaces",
  {
    id: bigint("id", { mode: "number", unsigned: true })
      .primaryKey()
      .autoincrement(),
    handle: varchar("handle", { length: HANDLE_LENGTH }).notNull(),
    kind: varchar("kind", { length: CODE_LENGTH })
      .$type<NamespaceKind>()
      .notNull(),
    clerkUserId: varchar("clerk_user_id", { length: CLERK_ID_LENGTH }),
    clerkOrgId: varchar("clerk_org_id", { length: CLERK_ID_LENGTH }),
    claimedClerkUserId: varchar("claimed_clerk_user_id", {
      length: CLERK_ID_LENGTH,
    }),
    claimedClerkOrgId: varchar("claimed_clerk_org_id", {
      length: CLERK_ID_LENGTH,
    }),
    status: varchar("status", { length: CODE_LENGTH })
      .$type<NamespaceStatus>()
      .notNull(),
    activeOperationId: bigint("active_operation_id", {
      mode: "number",
      unsigned: true,
    }),
    createdAt: timestamp("created_at", { mode: "date", fsp: 3 })
      .default(sql`CURRENT_TIMESTAMP(3)`)
      .notNull(),
    updatedAt: timestamp("updated_at", { mode: "date", fsp: 3 })
      .default(sql`CURRENT_TIMESTAMP(3)`)
      .onUpdateNow()
      .notNull(),
  },
  (table) => ({
    handleUq: uniqueIndex("namespaces_handle_uq").on(table.handle),
    activeOperationUq: uniqueIndex("namespaces_active_operation_uq").on(
      table.activeOperationId
    ),
    claimedUserUq: uniqueIndex("namespaces_claimed_user_uq").on(
      table.claimedClerkUserId
    ),
    claimedOrgUq: uniqueIndex("namespaces_claimed_org_uq").on(
      table.claimedClerkOrgId
    ),
    userIdx: index("namespaces_user_idx").on(
      table.clerkUserId,
      table.status
    ),
    orgIdx: index("namespaces_org_idx").on(table.clerkOrgId, table.status),
  })
);

export const namespaceOperations = mysqlTable(
  "lightfast_namespace_operations",
  {
    id: bigint("id", { mode: "number", unsigned: true })
      .primaryKey()
      .autoincrement(),
    operationType: varchar("operation_type", { length: CODE_LENGTH })
      .$type<NamespaceOperationType>()
      .notNull(),
    ownerKind: varchar("owner_kind", { length: CODE_LENGTH })
      .$type<NamespaceKind>()
      .notNull(),
    clerkUserId: varchar("clerk_user_id", { length: CLERK_ID_LENGTH }),
    clerkOrgId: varchar("clerk_org_id", { length: CLERK_ID_LENGTH }),
    idempotencyClerkUserId: varchar("idempotency_clerk_user_id", {
      length: CLERK_ID_LENGTH,
    }),
    idempotencyClerkOrgId: varchar("idempotency_clerk_org_id", {
      length: CLERK_ID_LENGTH,
    }),
    fromHandle: varchar("from_handle", { length: HANDLE_LENGTH }),
    toHandle: varchar("to_handle", { length: HANDLE_LENGTH }).notNull(),
    status: varchar("status", { length: CODE_LENGTH })
      .$type<NamespaceOperationStatus>()
      .notNull(),
    idempotencyKey: varchar("idempotency_key", {
      length: IDEMPOTENCY_KEY_LENGTH,
    }).notNull(),
    errorCode: varchar("error_code", { length: ERROR_CODE_LENGTH }),
    errorMessage: varchar("error_message", { length: ERROR_MESSAGE_LENGTH }),
    createdAt: timestamp("created_at", { mode: "date", fsp: 3 })
      .default(sql`CURRENT_TIMESTAMP(3)`)
      .notNull(),
    updatedAt: timestamp("updated_at", { mode: "date", fsp: 3 })
      .default(sql`CURRENT_TIMESTAMP(3)`)
      .onUpdateNow()
      .notNull(),
    expiresAt: timestamp("expires_at", { mode: "date", fsp: 3 }),
  },
  (table) => ({
    userIdempotencyUq: uniqueIndex(
      "namespace_operations_user_idempotency_uq"

      table.idempotencyClerkUserId,
      table.operationType,
      table.idempotencyKey

      table.idempotencyClerkOrgId,
      table.operationType,
      table.idempotencyKey

    statusIdx: index("namespace_operations_status_idx").on(
      table.status,

    userIdx: index("namespace_operations_user_idx").on(
      table.clerkUserId,
      table.status
    ),
    orgIdx: index("namespace_operations_org_idx").on(
      table.clerkOrgId,
      table.status
    ),
  })
);

export type Namespace = typeof namespaces.$inferSelect;
```

```ts
export type NamespaceConflictCode =

  | "OWNER_NAMESPACE_IN_PROGRESS"

export class NamespaceConflictError extends Error {
  constructor(
    public readonly code: NamespaceConflictCode,
    message: string
```

```ts
import type { NamespaceOperationStatus } from "../../schema";
import { NamespaceOperationTransitionError } from "./errors";

export type NamespaceOperationEvent =
  | { type: "RESERVE_NAMESPACE" }
  | { type: "MARK_CLERK_APPLIED" }
  | { type: "DELETE_PRE_CLERK_RESERVATION" }
  | { type: "FINALIZE" }
  | { type: "COMPENSATE" }
  | { type: "FAIL" };

export const namespaceOperationMachine = createMachine({
  id: "namespaceOperation",
  initial: "started",
  states: {
    started: {
      on: {
        FAIL: { target: "failed" },
        RESERVE_NAMESPACE: { target: "namespace_reserved" },
      },
    },
    namespace_reserved: {
      on: {
        DELETE_PRE_CLERK_RESERVATION: { target: "failed" },
        MARK_CLERK_APPLIED: { target: "clerk_applied" },
      },
    },
    clerk_applied: {
      on: {
        FINALIZE: { target: "finalized" },
        COMPENSATE: { target: "compensating" },
      },
    },
    compensating: {
      on: {
        FINALIZE: { target: "finalized" },
        FAIL: { target: "failed" },
      },
    },
    finalized: {
      type: "final",
    },
    failed: {
      type: "final",
    },
  },
});

export function isTerminalNamespaceOperationStatus(
  status: NamespaceOperationStatus
) {
  return status === "finalized" || status === "failed";
}

export function getNextNamespaceOperationStatus(
  status: NamespaceOperationStatus,
  event: NamespaceOperationEvent
): NamespaceOperationStatus {
  const current = namespaceOperationMachine.resolveState({ value: status });
  const [next] = transition(namespaceOperationMachine, current, event);
  const nextStatus = next.value;

  if (typeof nextStatus !== "string" || nextStatus === status) {
    throw new NamespaceOperationTransitionError(status, event);
  }

  return nextStatus as NamespaceOperationStatus;
}

export function canTransitionNamespaceOperation(
  status: NamespaceOperationStatus,
  event: NamespaceOperationEvent
) {
  try {
    getNextNamespaceOperationStatus(status, event);
    return true;
  } catch (error) {
    if (error instanceof NamespaceOperationTransitionError) {
      return false;
    }
    throw error;
  }
}
```

Idempotency keys are scoped to `(owner, operation_type)`, not global. The operation table uses nullable owner mirror columns plus separate user/org unique indexes so two different users or organizations can safely reuse the same caller-generated idempotency key.

Idempotency replay is sticky for terminal states. Reusing the same idempotency key returns the existing operation whether it is `started`, `namespace_reserved`, `clerk_applied`, `finalized`, or `failed`; it never creates a new operation or retries a failed reservation. A caller that wants a new attempt after conflict resolution must provide a new idempotency key.

`expires_at` is nullable because not every operation type is necessarily expirable, especially backfill operations. `startNamespaceOperation` must set it for normal create operations; later plans can split expirable and non-expirable operation constructors if that distinction needs stronger typing.

Owner columns are physically nullable but semantically required by `owner_kind`. For `owner_kind = "user"`, `clerk_user_id` and `idempotency_clerk_user_id` must be present. For `owner_kind = "org"`, `clerk_org_id` and `idempotency_clerk_org_id` must be present. Enforce this in TypeScript constructors and helpers rather than DB check constraints to keep the schema Vitess-friendly.

```ts
import { and, eq } from "drizzle-orm";
import { lightfastHandleSchema } from "@repo/app-validation";
import type { Database } from "../../client";
import type {
  Namespace,
  NamespaceKind,
  NamespaceOperation,
  NamespaceOperationType,
} from "../../schema";
import { namespaceOperations, namespaces } from "../../schema";
import { isDuplicateKeyError } from "../drizzle-results";
import {
  NamespaceConflictError,
  NamespaceOperationConcurrencyError,
  type NamespaceConflictCode,
} from "./errors";

export interface StartNamespaceOperationInput {
  clerkOrgId?: string | null;
  clerkUserId?: string | null;
  idempotencyKey: string;
  operationType: NamespaceOperationType;
  ownerKind: NamespaceKind;
  toHandle: string;
}

type NamespaceOperationIdempotencyInput = Pick<
  StartNamespaceOperationInput,
  "clerkOrgId" | "clerkUserId" | "idempotencyKey" | "operationType" | "ownerKind"
>;

export async function getNamespaceOperationByIdempotencyKey(
  db: Database,
  input: NamespaceOperationIdempotencyInput
): Promise<NamespaceOperation | undefined> {
  const ownerCondition =
    input.ownerKind === "user"
      ? eq(
          namespaceOperations.idempotencyClerkUserId,
          getRequiredClerkUserId(input)
        )
      : eq(
          namespaceOperations.idempotencyClerkOrgId,
          getRequiredClerkOrgId(input)
        );
  const [row] = await db
    .select()
    .from(namespaceOperations)
    .where(
      and(
        eq(namespaceOperations.idempotencyKey, input.idempotencyKey),
        eq(namespaceOperations.operationType, input.operationType),
        eq(namespaceOperations.ownerKind, input.ownerKind),
        ownerCondition
      )
    )
    .limit(1);
  return row;
}

export async function getNamespaceOperationById(
  db: Database,
  id: number
): Promise<NamespaceOperation | undefined> {
  const [row] = await db
    .select()
    .from(namespaceOperations)
    .where(eq(namespaceOperations.id, id))
    .limit(1);
  return row;
}

export async function getNamespaceByHandle(
  db: Database,
  handle: string
): Promise<Namespace | undefined> {
  const normalized = lightfastHandleSchema.parse(handle);
  const [row] = await db
    .select()
    .from(namespaces)
    .where(eq(namespaces.handle, normalized))
    .limit(1);
  return row;
}

export async function getClaimedNamespaceForOwner(
  db: Database,
  operation: Pick<NamespaceOperation, "ownerKind" | "clerkOrgId" | "clerkUserId">
): Promise<Namespace | undefined> {
  if (operation.ownerKind === "user" && operation.clerkUserId) {
    const [row] = await db
      .select()
      .from(namespaces)
      .where(eq(namespaces.claimedClerkUserId, operation.clerkUserId))
      .limit(1);
    return row;
  }

  if (operation.ownerKind === "org" && operation.clerkOrgId) {
    const [row] = await db
      .select()
      .from(namespaces)
      .where(eq(namespaces.claimedClerkOrgId, operation.clerkOrgId))
      .limit(1);
    return row;
  }

  return undefined;
}

export async function startNamespaceOperation(
  db: Database,
  input: StartNamespaceOperationInput
): Promise<NamespaceOperation> {
  const toHandle = lightfastHandleSchema.parse(input.toHandle);
  const expiresAt = new Date(Date.now() + 15 * 60 * 1000);
  const idempotencyOwner = getIdempotencyOwner(input);

  try {
    const inserted = await db
      .insert(namespaceOperations)
      .values({
        clerkOrgId: input.clerkOrgId ?? null,
        clerkUserId: input.clerkUserId ?? null,
        expiresAt,
        idempotencyClerkOrgId: idempotencyOwner.clerkOrgId,
        idempotencyClerkUserId: idempotencyOwner.clerkUserId,
        idempotencyKey: input.idempotencyKey,
        operationType: input.operationType,
        ownerKind: input.ownerKind,
        status: "started",
        toHandle,
      })
      .$returningId();

    const id = inserted[0]?.id;
    if (typeof id !== "number") {
      throw new Error("Failed to create namespace operation");
    }

    const operation = await getNamespaceOperationById(db, id);
    if (!operation) {
      throw new Error(`Failed to load namespace operation ${id}`);
    }
    return operation;
  } catch (error) {
    if (!isDuplicateKeyError(error)) {
      throw error;
    }
    const existing = await getNamespaceOperationByIdempotencyKey(
      db,
      input
    );
    if (!existing) {
      throw error;
    }
    assertSameOperationInput(existing, {
      ...input,
      toHandle,
    });
    return existing;
  }
}

function getIdempotencyOwner(input: NamespaceOperationIdempotencyInput) {
  if (input.ownerKind === "user") {
    return {
      clerkOrgId: null,
      clerkUserId: getRequiredClerkUserId(input),
    };
  }

  return {
    clerkOrgId: getRequiredClerkOrgId(input),
    clerkUserId: null,
  };
}

function getRequiredClerkUserId(input: NamespaceOperationIdempotencyInput) {
  if (!input.clerkUserId) {
    throw new Error("User namespace operations require clerkUserId");
  }
  return input.clerkUserId;
}

function getRequiredClerkOrgId(input: NamespaceOperationIdempotencyInput) {
  if (!input.clerkOrgId) {
    throw new Error("Org namespace operations require clerkOrgId");
  }
  return input.clerkOrgId;
}

function assertSameOperationInput(
  existing: NamespaceOperation,
  input: StartNamespaceOperationInput
) {
  const same =
    existing.operationType === input.operationType &&
    existing.ownerKind === input.ownerKind &&
    existing.toHandle === input.toHandle &&
    (existing.clerkUserId ?? null) === (input.clerkUserId ?? null) &&
    (existing.clerkOrgId ?? null) === (input.clerkOrgId ?? null);

  if (!same) {
    throw new NamespaceConflictError(
      "IDEMPOTENCY_KEY_REUSED",
      "This namespace operation key was already used with different input"
    );
  }
}
```

Reservation idempotency is intentionally narrow. `startNamespaceOperation` may return an existing operation for the same idempotency key, and `reserveNamespaceForOperation` may recover only when the namespace row already points to that same operation id. A same-owner collision owned by a different operation id is a conflict, even when the requested handle is identical.
`reserveNamespaceForOperation` only emits the `RESERVE_NAMESPACE` event when the operation is exactly `started`. Replays for `namespace_reserved`, `clerk_applied`, `compensating`, `finalized`, or `failed` return the operation unchanged before touching the database. Repairing a missing namespace row for an already-advanced operation is a reconciliation concern, not a reservation helper side effect.
`reserveNamespaceForOperation` does not enforce `expires_at`. Expiration is only for reconciliation and safe cleanup of stuck `namespace_reserved` operations; immediate reservation should remain a deterministic DB mutex operation.
Reservation conflicts are terminal for the operation that attempted the reservation. The service must transition the operation to `failed` with the same conflict code, allow that transaction to commit, and only then throw `NamespaceConflictError` to the caller. Do not throw from inside the transaction after writing the failed state, because that would roll the failed state back.
`transitionNamespaceOperation` is a strict compare-and-set primitive. If the status update affects zero rows, it must throw `NamespaceOperationConcurrencyError`; higher-level helpers are responsible for explicit idempotent replay behavior.

```ts
type NamespaceReservationResult =
  | { ok: true; operation: NamespaceOperation }
  | { ok: false; code: NamespaceConflictCode; message: string };

export async function reserveNamespaceForOperation(
  db: Database,
  operation: NamespaceOperation
): Promise<NamespaceOperation> {
  if (operation.status !== "started") {
    return operation;
  }

  const result = await db.transaction(
    async (tx): Promise<NamespaceReservationResult> => {
    try {
      await tx
        .insert(namespaces)
        .values({
          activeOperationId: operation.id,
          clerkOrgId: operation.clerkOrgId,
          clerkUserId: operation.clerkUserId,
          claimedClerkOrgId:
            operation.ownerKind === "org" ? operation.clerkOrgId : null,
          claimedClerkUserId:
            operation.ownerKind === "user" ? operation.clerkUserId : null,
          handle: operation.toHandle,
          kind: operation.ownerKind,
          status: "reserved",
        })
        .$returningId();
    } catch (error) {
      if (!isDuplicateKeyError(error)) {
        throw error;
      }
      const existing = await getNamespaceByHandle(tx as Database, operation.toHandle);
      if (
        existing &&
        isSameNamespaceOwner(existing, operation) &&
        existing.activeOperationId === operation.id
      ) {
        return {
          ok: true,
          operation: await transitionNamespaceOperation(
            tx as Database,
            operation,
            { type: "RESERVE_NAMESPACE" }
          ),
        };
      }
      if (existing && isSameNamespaceOwner(existing, operation)) {
        return await failReservationWithConflict(
          tx as Database,
          operation,
          existing.status === "reserved"
            ? "OWNER_NAMESPACE_IN_PROGRESS"
            : "OWNER_ALREADY_CLAIMED",
          existing.status === "reserved"
            ? "Owner already has a namespace operation in progress"
            : "Owner already has a claimed namespace handle"
        );
      }
      if (existing) {
        return await failReservationWithConflict(
          tx as Database,
          operation,
          "HANDLE_ALREADY_CLAIMED",
          `Handle ${operation.toHandle} is already claimed`
        );
      }

      const ownerNamespace = await getClaimedNamespaceForOwner(
        tx as Database,
        operation
      );
      if (ownerNamespace) {
        return await failReservationWithConflict(
          tx as Database,
          operation,
          ownerNamespace.status === "reserved"
            ? "OWNER_NAMESPACE_IN_PROGRESS"
            : "OWNER_ALREADY_CLAIMED",
          ownerNamespace.status === "reserved"
            ? "Owner already has a namespace operation in progress"
            : "Owner already has a claimed namespace handle"
        );
      }
      return await failReservationWithConflict(
        tx as Database,
        operation,
        "HANDLE_ALREADY_CLAIMED",
        `Handle ${operation.toHandle} is already claimed`
      );
    }

    return {
      ok: true,
      operation: await transitionNamespaceOperation(
        tx as Database,
        operation,
        { type: "RESERVE_NAMESPACE" }
      ),
    };
  });

  if (!result.ok) {
    throw new NamespaceConflictError(result.code, result.message);
  }

  return result.operation;
}

async function failReservationWithConflict(
  db: Database,
  operation: NamespaceOperation,
  code: NamespaceConflictCode,
  message: string
): Promise<NamespaceReservationResult> {
  await transitionNamespaceOperation(
    db,
    operation,
    { type: "FAIL" },
    {
      errorCode: code,
      errorMessage: message,
    }
  );
  return { ok: false, code, message };
}

function isSameNamespaceOwner(
  namespace: Namespace,
  operation: NamespaceOperation
) {
  return (
    namespace.kind === operation.ownerKind &&
    (namespace.clerkUserId ?? null) === (operation.clerkUserId ?? null) &&
    (namespace.clerkOrgId ?? null) === (operation.clerkOrgId ?? null)
  );
}

export async function transitionNamespaceOperation(
  db: Database,
  operation: NamespaceOperation,
  event: NamespaceOperationEvent,
  patch: Partial<Pick<NamespaceOperation, "errorCode" | "errorMessage">> = {}
): Promise<NamespaceOperation> {
  const nextStatus = getNextNamespaceOperationStatus(operation.status, event);
  const result = await db
    .update(namespaceOperations)
    .set({
      ...patch,
      status: nextStatus,
    })
    .where(
      and(
        eq(namespaceOperations.id, operation.id),
        eq(namespaceOperations.status, operation.status)
      )
    );

  if (getRowsAffected(result) === 0) {
    throw new NamespaceOperationConcurrencyError(
      operation.id,
      operation.status,
      event
    );
  }

  const updated = await getNamespaceOperationById(db, operation.id);
  if (!updated) {
    throw new Error(`Failed to load namespace operation ${operation.id}`);
  }
  return updated;
}
```

`markNamespaceOperationClerkApplied` must only be called after a successful Clerk write response or after reconciliation performs a fresh Clerk read proving the owner currently has `toHandle`. Sending a Clerk request is not enough. Clerk timeouts and unknown responses leave the operation at `namespace_reserved` for reconciliation.

`finalizeNamespaceOperation` must activate the namespace row and clear `active_operation_id` before transitioning the operation to `finalized`, in the same DB transaction. If the operation transition fails, the namespace activation must roll back; there must never be a `finalized` operation whose namespace row is still `reserved`.

`failUnreservedNamespaceOperation` is only for operations that are still `started` and have no namespace row. Once an operation reaches `namespace_reserved`, plain failure is illegal because it would leave the handle and owner mutex reserved. Reserved cleanup must go through `deletePreClerkNamespaceReservation`, and that helper must use the explicit `DELETE_PRE_CLERK_RESERVATION` state-machine event.

`deletePreClerkNamespaceReservation` is a narrow cleanup primitive. It is only valid when the operation is still `namespace_reserved`, the namespace row still points at that operation, `expiresAt` has passed, and the caller has a fresh Clerk read proving the target user/org does not currently hold `toHandle`. Clerk read failures or timeouts must keep the reservation in place and retry reconciliation later. This foundation plan does not add that Clerk proof checker; the later reconciliation plan must provide it before using this helper.

```ts
export async function markNamespaceOperationClerkApplied(
  db: Database,
  operation: NamespaceOperation
): Promise<NamespaceOperation> {
  // Call only after Clerk confirms the handle, or reconciliation proves it.
  return await transitionNamespaceOperation(db, operation, {
    type: "MARK_CLERK_APPLIED",
  });
}

export async function finalizeNamespaceOperation(
  db: Database,
  operation: NamespaceOperation
): Promise<NamespaceOperation> {
  return await db.transaction(async (tx) => {
    // Activate first so a failed operation transition rolls back the namespace.
    const namespaceResult = await tx
      .update(namespaces)
      .set({
        activeOperationId: null,
        status: "active",
      })
      .where(
        and(
          eq(namespaces.handle, operation.toHandle),
          eq(namespaces.activeOperationId, operation.id),
          eq(namespaces.status, "reserved")
        )
      );

    if (getRowsAffected(namespaceResult) === 0) {
      throw new Error(
        `Failed to activate namespace ${operation.toHandle} for operation ${operation.id}`
      );
    }

    return await transitionNamespaceOperation(tx as Database, operation, {
      type: "FINALIZE",
    });
  });
}

export async function failUnreservedNamespaceOperation(
  db: Database,
  operation: NamespaceOperation,
  input: { errorCode: string; errorMessage: string }
): Promise<NamespaceOperation> {
  if (operation.status !== "started") {
    throw new Error("Only started namespace operations can be failed directly");
  }

  return await transitionNamespaceOperation(
    db,
    operation,
    { type: "FAIL" },
    {
      errorCode: input.errorCode,
      errorMessage: input.errorMessage,
    }
  );
}

export async function deletePreClerkNamespaceReservation(
  db: Database,
  operation: NamespaceOperation,
  input: { errorCode: string; errorMessage: string }
): Promise<NamespaceOperation> {
  if (operation.status !== "namespace_reserved") {
    throw new Error(
      "Only namespace_reserved operations can delete a pre-Clerk namespace reservation"
    );
  }

  return await db.transaction(async (tx) => {
    const failed = await transitionNamespaceOperation(
      tx as Database,
      operation,
      { type: "DELETE_PRE_CLERK_RESERVATION" },
      {
        errorCode: input.errorCode,
        errorMessage: input.errorMessage,
      }
    );

    if (failed.status !== "failed") {
      throw new Error(
        `Failed to mark namespace operation ${operation.id} failed before deleting reservation`
      );
    }

    const result = await tx
      .delete(namespaces)
      .where(
        and(
          eq(namespaces.handle, operation.toHandle),
          eq(namespaces.activeOperationId, operation.id),
          eq(namespaces.status, "reserved")
        )
      );

    if (getRowsAffected(result) === 0) {
      throw new Error(
        `Failed to delete namespace reservation ${operation.toHandle} for operation ${operation.id}`
      );
    }

    return failed;
  });
}
```
